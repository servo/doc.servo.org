enum Frame<'a> {
    Repetition(&'a Repetition),
    Capture(&'a Capture),
    Concat {
        head: &'a Hir,
        tail: &'a [Hir],
    },
    Alternation {
        head: &'a Hir,
        tail: &'a [Hir],
    },
}
Expand description

Represents a single stack frame while performing structural induction over an Hir.
