pub static mut proxy_handlers: [*const libc::c_void; 8] = [0 as *const libc::c_void; 8];
pub fn RegisterProxyHandlers() {
    unsafe {
        proxy_handlers[Proxies::CSSStyleDeclaration as usize] = codegen::Bindings::CSSStyleDeclarationBinding::DefineProxyHandler();
        proxy_handlers[Proxies::DOMRectList as usize] = codegen::Bindings::DOMRectListBinding::DefineProxyHandler();
        proxy_handlers[Proxies::DOMStringMap as usize] = codegen::Bindings::DOMStringMapBinding::DefineProxyHandler();
        proxy_handlers[Proxies::DOMTokenList as usize] = codegen::Bindings::DOMTokenListBinding::DefineProxyHandler();
        proxy_handlers[Proxies::HTMLCollection as usize] = codegen::Bindings::HTMLCollectionBinding::DefineProxyHandler();
        proxy_handlers[Proxies::NamedNodeMap as usize] = codegen::Bindings::NamedNodeMapBinding::DefineProxyHandler();
        proxy_handlers[Proxies::NodeList as usize] = codegen::Bindings::NodeListBinding::DefineProxyHandler();
        proxy_handlers[Proxies::Storage as usize] = codegen::Bindings::StorageBinding::DefineProxyHandler();
    }
}
