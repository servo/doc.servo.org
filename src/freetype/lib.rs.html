#[cfg(feature = "freetype-sys")]
extern crate freetype_sys;
extern crate libc;

pub type FT_Error = ::std::os::raw::c_int;

#[inline]
pub fn succeeded(error: FT_Error) -> bool {
    error == freetype::FT_Err_Ok as FT_Error
}
