pub struct RefCell<T: ?Sized> {
    pub(crate) borrows: RefCell<BorrowData>,
    pub(crate) inner: RefCell<T>,
}
Expand description

A RefCell that tracks outstanding borrows and reports stack traces for dynamic borrow failures.
