Writes to an owned, growable byte vector that supports seeking.

Example

extern crate rbml; fn main() { #![allow(unused_must_use)] use rbml::io::SeekableMemWriter; let mut w = SeekableMemWriter::new(); w.write(&[0, 1, 2]); assert_eq!(w.unwrap(), [0, 1, 2]); }
use rbml::io::SeekableMemWriter;

let mut w = SeekableMemWriter::new();
w.write(&[0, 1, 2]);

assert_eq!(w.unwrap(), [0, 1, 2]);
