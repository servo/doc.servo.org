The log-normal distribution ln N(mean, std_dev**2).

If X is log-normal distributed, then ln(X) is N(mean, std_dev**2) distributed.

Example

fn main() { use std::rand; use std::rand::distributions::{LogNormal, IndependentSample}; // mean 2, standard deviation 3 let log_normal = LogNormal::new(2.0, 3.0); let v = log_normal.ind_sample(&mut rand::thread_rng()); println!("{} is from an ln N(2, 9) distribution", v) }
use std::rand;
use std::rand::distributions::{LogNormal, IndependentSample};

// mean 2, standard deviation 3
let log_normal = LogNormal::new(2.0, 3.0);
let v = log_normal.ind_sample(&mut rand::thread_rng());
println!("{} is from an ln N(2, 9) distribution", v)
