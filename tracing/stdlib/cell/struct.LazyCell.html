Expand description

A value which is initialized on the first access.

For a thread-safe version of this struct, see std::sync::LazyLock.

ยงExamples

use std::cell::LazyCell;

let lazy: LazyCell<i32> = LazyCell::new(|| {
    println!("initializing");
    92
});
println!("ready");
println!("{}", *lazy);
println!("{}", *lazy);

// Prints:
//   ready
//   initializing
//   92
//   92
