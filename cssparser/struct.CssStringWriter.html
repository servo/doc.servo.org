A TextWriter adaptor that escapes text for writing as a double-quoted CSS string. Quotes are not included.

Typical usage:

fn write_foo<W>(foo: &Foo, dest: &mut W) -> text_writer::Result where W: TextWriter {
    try!(dest.write_char('"'));
    {
        let mut string_dest = CssStringWriter::new(dest);
        // Write into string_dest...
    }
    try!(dest.write_char('"'));
    Ok(())
}
