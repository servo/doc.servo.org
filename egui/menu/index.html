Module egui::menu

source ·
Expand description

Menu bar functionality (very basic so far).

Usage:

fn show_menu(ui: &mut egui::Ui) {
    use egui::{menu, Button};

    menu::bar(ui, |ui| {
        ui.menu_button("File", |ui| {
            if ui.button("Open").clicked() {
                // …
            }
        });
    });
}
